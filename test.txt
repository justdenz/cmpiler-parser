main(){
    String line = "";
    String col_header = "          ";
    int size;
    scan("Input size: ", size);
    int x = 1;
    int y = 1;

    while x up to size {
	col_header = col_header + x + "	  ";
    }
    print(col_header);
    x = 1;
    while x up to size{
        line = x + " ";
	y = 1;
        while y up to size {
            int prod;
            prod = x * y;
	    line = line + "	  " + prod;
        }
        print(line);
	line = "";
    }
}
