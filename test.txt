func void printF(int size) {

    int mid = size / 2;

    // for upper bar
    String s = "";
    int i = 0;
    while i up to size {
        s = s + "*";
    }
    print(s);

    for int x = 0 up to mid - 1 {
        print("*");
    }

    String m = "";
    i = 0;
    while i up to size - 1 {
        m = m + "*";
    }

    print(m);
    
    for int y = 0 up to size - mid - 1 {
        print("*");
    }    
    
}
    
func void printT(int size) {

    int mid = size / 2;
    
    // centering
    String spaces = "";
    for int space = 0 up to mid {
        spaces = spaces + " ";
    }

    int j = 0;
    String s = "";
    while j up to size {
        s = s + "*";
    }

    print(s);

    int i = 0;
    while i up to size-1 {
        print(spaces + "*");
    }
}

func void printL(int size) {
    
    int i = 0;
    while i up to size-1 {
        print("*");
    }

    int j = 0;
    String s = "";
    while j up to size {
        s = s + "*";
    }

    print(s);
}

main() {
    
    String c;
    scan("Enter char: ", c);

    if (c == "F") then {
        int size;
        scan("Enter font size: ", size);

        if (size > 4)then {
            printF(size);
        } else then {
            print("INVALID! Font size should be at least 5.");
        }


    } else if (c == "T") then {

        int size;
        scan("Enter font size: ", size);

        if (size > 4)then {
            printT(size);
        } else then {
            print("INVALID! Font size should be at least 5.");
        }


    } else if (c == "L") then {

        int size;
        scan("Enter font size: ", size);

        if (size > 4)then {
            printL(size);
        } else then {
            print("INVALID! Font size should be at least 5.");
        }

    } else then {
        print("INVALID! Only supports 'FTL'. Please try again");
    }
}
